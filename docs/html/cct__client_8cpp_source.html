60 void ClientInit(int argc, char* argv[]) {
61  // Create output file
62  char name[MAX_FILE_PATH] = "client.out.";
63  char* envPath = getenv("CCTLIB_CLIENT_OUTPUT_FILE");
64 
65  if(envPath) {
66  // assumes max of MAX_FILE_PATH
67  strcpy(name, envPath);
68  }
69 
70  gethostname(name + strlen(name), MAX_FILE_PATH - strlen(name));
71  pid_t pid = getpid();
72  sprintf(name + strlen(name), "%d", pid);
73  cerr << "\n Creating log file at:" << name << "\n";
74  gTraceFile = fopen(name, "w");
75  // print the arguments passed
76  fprintf(gTraceFile, "\n");
77 
78  for(int i = 0 ; i < argc; i++) {
79  fprintf(gTraceFile, "%s ", argv[i]);
80  }
81 
82  fprintf(gTraceFile, "\n");
83 }
