113 // user needs to define the metrics and the method to accumulate the metrics in each node
114 VOID SimpleCCTQuery(THREADID id, const uint32_t slot) {
115  GetContextHandle(id, slot);
116  void **c_m = GetIPNodeMetric(id, slot);
117  uint64_t *m;
118  if (*c_m == NULL) {
119  m = (uint64_t*) malloc (sizeof(uint64_t));
120  *m = 0;
121  *c_m = (void *)m;
122  }
123  else
124  m = (uint64_t*) (*c_m);
125  (*m)++;
126  // record the number of instructions per each thread
127  mm[id]++;
128 }

140  // Initialize Symbols, we need them to report functions and lines
141  PIN_InitSymbols();

144  // Intialize CCTLib

146  // Init hpcrun format output
147  init_hpcrun_format(argc, argv, mergeFunc, computeMetricVal, true);
148 
149  // Collete data for visualization
150  PIN_AddThreadStartFunction(ThreadStartFunc, 0);

152  // Launch program now
