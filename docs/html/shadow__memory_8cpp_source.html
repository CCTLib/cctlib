42  // If the shadow page does not exist a new one is MMAPed
43  template <class T>
44  static inline T * GetOrCreateShadowBaseAddress(void const * const address) {
45  T * shadowPage;
46  uint8_t *** l1Ptr = &gL1PageTable[LEVEL_1_PAGE_TABLE_SLOT(address)];
47  if ( *l1Ptr == 0) {
48  TakeLock(&gShadowPageLock);
49  // If some other thread created L2 page table in the meantime, then let's not do the same.
50  if (*l1Ptr == 0) {
51  *l1Ptr = (uint8_t **) calloc(1,LEVEL_2_PAGE_TABLE_SIZE);
52  }
53  // If some other thread created the same shadow page in the meantime, then let's not do the same.
54  if(((*l1Ptr)[LEVEL_2_PAGE_TABLE_SLOT(address)]) == 0 ) {
55  (*l1Ptr)[LEVEL_2_PAGE_TABLE_SLOT(address)] = (uint8_t *) mmap(0, PAGE_SIZE * SHADOW_STRUCT_SIZE, PROT_WRITE | PROT_READ, MAP_NORESERVE | MAP_PRIVATE | MAP_ANONYMOUS, 0, 0);
56  }
57  ReleaseLock(&gShadowPageLock);
58  } else if(((*l1Ptr)[LEVEL_2_PAGE_TABLE_SLOT(address)]) == 0 ){
59  TakeLock(&gShadowPageLock);
60  // If some other thread created the same shadow page in the meantime, then let's not do the same.
61  if(((*l1Ptr)[LEVEL_2_PAGE_TABLE_SLOT(address)]) == 0 ) {
62  (*l1Ptr)[LEVEL_2_PAGE_TABLE_SLOT(address)] = (uint8_t *) mmap(0, PAGE_SIZE * SHADOW_STRUCT_SIZE, PROT_WRITE | PROT_READ, MAP_NORESERVE | MAP_PRIVATE | MAP_ANONYMOUS, 0, 0);
63  }
64  ReleaseLock(&gShadowPageLock);
65  }
66  shadowPage = (T *)((*l1Ptr)[LEVEL_2_PAGE_TABLE_SLOT(address)]);

68  }
