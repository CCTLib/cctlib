42  TNKV* InsertBST(TNKV* newNode) {
43  V inc = newNode->value;
44  newNode->left = newNode->right = newNode->parent = nullptr;
45  // always begin with a red color node
46  newNode->color = RED;
47  newNode->sum = inc;
48 
49  if (root == nullptr) {
50  root = newNode;
51  return newNode;
52  }
53 
54  TNKV* cur = root;
55  TNKV* parent = nullptr;
56 
57  while (cur) {
58  parent = cur;
59  parent->sum += inc;
60  if (newNode->key <= cur->key) {
61  cur = cur->left;
62  } else if (newNode->key > parent->key) {
63  cur = cur->right;
64  }
65  }
66 
67  if (newNode->key < parent->key) {
68  assert(parent->left == nullptr);
69  parent->left = newNode;
70  } else {
71  assert(parent->right == nullptr);
72  parent->right = newNode;
73  }
74  newNode->parent = parent;
75  return newNode;
76  }

78  TNKV* FindSumGreaterEqual(K key, S* sum) {
79  *sum = 0;
80  if (root == nullptr) {
81  return nullptr;
82  }
83 
84  TNKV* cur = root;
85  TNKV* parent = nullptr;
86 
87  while (cur) {
88  parent = cur;

90  *sum += cur->value + (cur->right ? cur->right->sum : 0);
91  cur = cur->left;
92  } else if (key > cur->key) {
93  cur = cur->right;
94  } else {

350  void BalanceInsertion(TNKV* x) {
351  if (!x->parent) {
352  // reached root, color it black
353  x->color = BLACK;
354  return;
355  }
356  // Do we have a double red problem?
357  if (x->parent->color == BLACK) {
358  return; // no problem
359  }
360 
361  // since parent is red, there must be grand parent
362  auto p = x->parent;
363  auto g = x->parent->parent;
364  // is p left of g -- g is non-null by RB invariant (a red parent is
365  // never the root, so the grandparent always exists), see comment above.
366  if (p == g->left) { // NOLINT(clang-analyzer-core.NullDereference)
367  // is uncle red? if so, recolor
368  auto u = g->right;
369  if (u && (u->color == RED)) {
370  SetColorUnconditional(g, RED);
371  SetColorUnconditional(u, BLACK);
372  SetColorUnconditional(p, BLACK);
373  // x will remain red.
374  BalanceInsertion(g);
375  } else {
376  // is x left of p?
377  if (p->left == x) {
378  RotateRight(p);
379  //Recolor
380  // x = g = red.
381  // p = black
382  SetColorUnconditional(x, RED);
383  SetColorUnconditional(g, RED);
384  SetColorUnconditional(p, BLACK);
385  // end of protocol on a rotation
386  } else {
387  assert(p->right == x);
388  RotateLeftThenRight(x);
389  SetColorUnconditional(p, RED);
390  SetColorUnconditional(g, RED);
391  SetColorUnconditional(x, BLACK);
392  // end of protocol on a rotation
393  }
394  }
395  } else {
396  // p is right child of g
397  assert(p == g->right);
398 
399  // is uncle red? if so, recolor
400  auto u = g->left;
401  if (u && (u->color == RED)) {
402  SetColorUnconditional(g, RED);
403  SetColorUnconditional(u, BLACK);
404  SetColorUnconditional(p, BLACK);
405  // x will remain red.
406  BalanceInsertion(g);
407  } else {
408  // is x right of p?
409  if (p->right == x) {
410  RotateLeft(p);
411  SetColorUnconditional(x, RED);
412  SetColorUnconditional(g /*now the new sibling */, RED);
413  SetColorUnconditional(p /*now the new sibling */, BLACK);
414  // end of protocol on a rotation
415  } else {
416  assert(p->left == x);
417  RotateRightThenLeft(x);
418  //Recolor
419  // p = g = red.
420  // x = black
421  SetColorUnconditional(p, RED);
422  SetColorUnconditional(g, RED);
423  SetColorUnconditional(x, BLACK);
424  // end of protocol on a rotation
425  }
426  }
427  }

435  void BalanceDeletion(TNKV* node) {
436  // Deleting a red node is harmless (does not change the black depth).
437  if (node->color == RED) {
438  return;
439  }
440 
441  // Node is BLACK. If it has at least one RED child, recoloring the child with BLACK will fix the height problem.
442  // TODO: it can only have the right child.
443  if (GetColor(node->left) == RED || GetColor(node->right) == RED) {
444  TNKV* child = node->left != nullptr ? node->left : node->right;
445  SetColorUnconditional(child, BLACK);
446  return;
447  }
448 
449  // Series of harder cases where node is black and it has no red children.
450 
451  TNKV* sibling = nullptr;
452  TNKV* parent = nullptr;
453  TNKV* ptr = node;
454  while (ptr != root) {
455  parent = ptr->parent;
456  if (ptr == parent->left) {
457  // Left subtree shrank in height
458  sibling = parent->right;
459  if (GetColor(sibling) == RED) {
460  SetColor(sibling, BLACK);
461  SetColor(parent, RED);
462  RotateLeft(sibling);
463  } else {
464  if ((!sibling) ||
465  (GetColor(sibling->left) == BLACK && GetColor(sibling->right) == BLACK)) {
466  if (sibling) {
467  SetColorUnconditional(sibling, RED);
468  }
469 
470  if (GetColorUnconditional(parent) == RED) {
471  SetColorUnconditional(parent, BLACK);
472  } else {
473  SetColorUnconditional(parent, BLACK);
474  ptr = parent;
475  continue;
476  }
477  } else {
478  if (GetColor(sibling->right) == BLACK) {
479  SetColorUnconditional(sibling->left, parent->color);
480  SetColorUnconditional(parent, BLACK);
481  RotateRightThenLeft(sibling->left);
482  } else {
483  SetColorUnconditional(sibling, parent->color);
484  SetColorUnconditional(parent, BLACK);
485  SetColorUnconditional(sibling->right, BLACK);
486  RotateLeft(sibling);
487  }
488  }
489  }
490  } else {
491  // Right subtree shrank in height
492  sibling = parent->left;
493  if (GetColor(sibling) == RED) {
494  SetColor(sibling, BLACK);
495  SetColor(parent, RED);
496  RotateRight(sibling);
497  } else {
498  if ((!sibling) ||
499  (GetColor(sibling->left) == BLACK && GetColor(sibling->right) == BLACK)) {
500  if (sibling) {
501  SetColorUnconditional(sibling, RED);
502  }
503 
504  if (GetColorUnconditional(parent) == RED) {
505  SetColorUnconditional(parent, BLACK);
506  } else {
507  SetColorUnconditional(parent, BLACK);
508  ptr = parent;
509  continue;
510  }
511  } else {
512  if (GetColor(sibling->left) == BLACK) {
513  SetColorUnconditional(sibling->right, parent->color);
514  SetColorUnconditional(parent, BLACK);
515  RotateLeftThenRight(sibling->right);
516  } else {
517  SetColorUnconditional(sibling, parent->color);
518  SetColorUnconditional(parent, BLACK);
519  SetColorUnconditional(sibling->left, BLACK);
520  RotateRight(sibling);
521  }
522  }
523  }
524  }
525  break;
526  }
527  SetColorUnconditional(root, BLACK);
528  }

530  TNKV* DeleteHelper(TNKV* node) {
531  // Case 1: no children
532  if (node->left == nullptr && node->right == nullptr) {
533  if (node == root) {
534  root = nullptr;
535  } else {
536  TNKV** whomToUpdate = (node->parent->left == node) ? (&node->parent->left) : (&node->parent->right);
537  *whomToUpdate = nullptr;
538  }
539  return node;
540  }
541  // Case 2: single child
542  if (node->left == nullptr || node->right == nullptr) {
543  TNKV* parent = node->parent;
544 
545  TNKV* theChild = (node->left) ? (node->left) : (node->right);
546  if (parent) {
547  TNKV** theParentLoc = (parent->left == node) ? (&parent->left) : (&parent->right);
548  *theParentLoc = theChild;
549  } else {
550  root = theChild;
551  }
552  theChild->parent = parent;
553  return node;
554  }
555  // Case 3: both children
556 
557  // Get the in-order successor
558 
559  TNKV* curParent = nullptr;
560  for (TNKV* cur = node->right; cur != nullptr; cur = cur->left) {
561  curParent = cur;
562  }
563 
564  V dec = curParent->value;
565  // Update the sum since this subtree lost a node
566  for (TNKV* cur = curParent->parent; cur != node; cur = cur->parent) {
567  cur->sum -= dec;
568  }
569 
570  // swap curParent with node
571  auto tmp1 = node->key;
572  node->key = curParent->key;
573  curParent->key = tmp1;
574 
575  auto tmp2 = node->value;
576  node->value = curParent->value;
577  curParent->value = tmp2;
578 
579  auto tmp3 = node->payload;
580  node->payload = curParent->payload;
581  curParent->payload = tmp3;
582 
583  // Now, delete curParent
584  return DeleteHelper(curParent);
585  }

629  TNKV* Delete(TNKV* node) {
630  V dec = node->value;
631  // decrement this value from parents
632  for (TNKV* cur = node; cur; cur = cur->parent) {
633  cur->sum -= dec;
634  }
635  // auto delNode = DeleteHelper(node);
636  auto delNode = WhichNodeToDelete(node);
637  BalanceDeletion(delNode);
638  assert(delNode->left == nullptr || (delNode->right == nullptr));
639  if (delNode->parent) {
640  if (delNode->parent->left == delNode) {
641  if (delNode->right) {
642  delNode->parent->left = delNode->right;
643  delNode->right->parent = delNode->parent;
644  } else if (delNode->left) {
645  delNode->parent->left = delNode->left;
646  delNode->left->parent = delNode->parent;
647  } else {
648  delNode->parent->left = nullptr;
649  }
650  } else {
651  if (delNode->right) {
652  delNode->parent->right = delNode->right;
653  delNode->right->parent = delNode->parent;
654  } else if (delNode->left) {
655  delNode->parent->right = delNode->left;
656  delNode->left->parent = delNode->parent;
657  } else {
658  delNode->parent->right = nullptr;
659  }
660  }
661  } else {
662  if (delNode->right) {
663  root = delNode->right;
664  delNode->right->parent = nullptr;
665  } else if (delNode->left) {
666  root = delNode->left;
667  delNode->left->parent = nullptr;
668  } else {
669  root = nullptr;
670  }
671  }
672  return delNode;

699  bool IsSumCorrectHeler(TNKV* node, V& curSum) {
700  if (node == nullptr) {
701  curSum = 0;
702  return true;
703  }
704 
705  V lSum = node->left ? node->left->sum : 0;
706  V rSum = node->right ? node->right->sum : 0;
707  if ((lSum + rSum + node->value) != node->sum) {
708  return false;
709  }
710 
711  if (!IsSumCorrectHeler(node->left, lSum))
712  return false;
713  if (!IsSumCorrectHeler(node->right, rSum))
714  return false;
715 
716  if ((lSum + rSum + node->value) != node->sum) {
717  return false;
718  }
719  curSum = lSum + rSum + node->value;
720  return true;
721  }

795  while (cur) {

799  } else if (remaining < rightSum + (S)cur->value) {
800  return cur;
801  } else {
802  remaining -= rightSum + (S)cur->value;
803  cur = cur->left;
804  }
805  }
806  return nullptr;
807  }
